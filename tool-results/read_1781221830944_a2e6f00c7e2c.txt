   312→    // ISO week: first week contains the year's first Thursday

   323→      // Only include weeks whose Monday is on or after June 1, 2026

   341→    // If target is current or future week, show current value

   525→            {progress > 0 && <circle cx={center + 8} cy={center + 8} r={radius - strokeWidth * 0.3} fill="none" stroke={color} strokeWidth={strokeWidth * 0.15} strokeDasharray={circumference * 0.95} strokeDashoffset={offset * 0.95} strokeLinecap="round" opacity="0.3" className="transition-all duration-1000" />}
